HMM with discrete observations

using Distributions
using HMMBase
using Plots

https://en.wikipedia.org/wiki/Viterbi_algorithm#Example

a = [0.6, 0.4]
A = [0.7 0.3; 0.4 0.6]
B = [Categorical([0.5, 0.4, 0.1]), Categorical([0.1, 0.3, 0.6])]
hmm = HMM(a, A, B)
HMM{Univariate,Float64}([0.6, 0.4], [0.7 0.3; 0.4 0.6], Distribution{Univariate,S} where S<:ValueSupport[DiscreteNonParametric{Int64,Float64,Base.OneTo{Int64},Array{Float64,1}}(support=Base.OneTo(3), p=[0.5, 0.4, 0.1]), DiscreteNonParametric{Int64,Float64,Base.OneTo{Int64},Array{Float64,1}}(support=Base.OneTo(3), p=[0.1, 0.3, 0.6])])
z, y = rand(hmm, 250)
z_viterbi = viterbi(hmm, y);
1-element Array{Int64,1}:
 1
plot(y, linetype=:steppre, label="Observations", size=(600,200))
plot(z, linetype=:steppre, label="True hidden state", size=(600,200))
plot(z_viterbi, linetype=:steppre, label="Viterbi decoded hidden state", size=(600,200))

Maximum Likelihood Estimation

using Distributions
using HMMBase
using Plots

y1 = rand(Normal(0,2), 1000)
y2 = rand(Normal(10,1), 500)
y = vcat(y1, y2, y1, y2)

plot(y, linetype=:steppre, size=(600,200))

For now HMMBase does not handle the initialization of the parameters. Hence we must instantiate an initial HMM by hand.

hmm = HMM([0.5 0.5; 0.5 0.5], [Normal(-1,2), Normal(15,2)])
hmm = fit_mle(hmm, y, display = :iter)
(HMM{Univariate,Float64}([1.0, 3.664139044720312e-64], [0.9989999998878044 0.0010000001121956157; 0.0010010012260593191 0.9989989987739407], Distribution{Univariate,S} where S<:ValueSupport[Normal{Float64}(μ=-0.012381083777039434, σ=1.9777106363761598), Normal{Float64}(μ=9.954701024895327, σ=0.9795830705580061)]), HMMBase.EMHistory(true, 3, [-5625.4685727648175, -5623.787987241749, -5623.787987241747]))
z_viterbi = viterbi(hmm, y)
plot(z_viterbi, linetype=:steppre, label="Viterbi decoded hidden state", size=(600,200))

Maximum Likelihood Estimation

using Distributions
using HMMBase
using Plots

y1 = rand(Normal(0,2), 1000)
y2 = rand(Normal(10,1), 500)
y = vcat(y1, y2, y1, y2)

plot(y, linetype=:steppre, size=(600,200))

For now HMMBase does not handle the initialization of the parameters. Hence we must instantiate an initial HMM by hand.

hmm = HMM([0.5 0.5; 0.5 0.5], [Normal(-1,2), Normal(15,2)])
hmm = fit_mle(hmm, y, display = :iter)
(HMM{Univariate,Float64}([1.0, 1.07145906211811e-50], [0.9990000000110563 0.0009999999889437508; 0.0010010012149329923 0.9989989987850669], Distribution{Univariate,S} where S<:ValueSupport[Normal{Float64}(μ=-0.06309116645997591, σ=1.9520285232425425), Normal{Float64}(μ=9.976306283135177, σ=0.9602347925760636)]), HMMBase.EMHistory(true, 3, [-5578.792738790851, -5577.696029182358, -5577.696029182345]))
z_viterbi = viterbi(hmm, y)
plot(z_viterbi, linetype=:steppre, label="Viterbi decoded hidden state", size=(600,200))

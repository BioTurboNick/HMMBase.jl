    MLE Estimator

    using Distributions
using HMMBase
using Plots

y1 = rand(Normal(0,2), 1000)
y2 = rand(Normal(10,1), 500)
y = vcat(y1, y2, y1, y2)

plot(y, linetype=:steppre, size=(600,200))

    For now HMMBase does not handle the initialization of the parameters. Hence we must instantiate an initial HMM by hand.

    hmm = HMM([0.5 0.5; 0.5 0.5], [Normal(-1,1), Normal(15,1)])
hmm, log_likelihood = fit_mle!(hmm, y, verbose=true)
    Converged after 4 iterations
(HMM{Univariate,Float64}([1.0, 9.88751e-132], [0.998875 0.00112473; 0.00125065 0.998749], Distribution{Univariate,S} where S<:ValueSupport[Normal{Float64}(μ=0.0831442, σ=1.94576), Normal{Float64}(μ=10.0182, σ=1.00859)]), -5620.43631330111)
    z_viterbi = viterbi(hmm, y)
plot(z_viterbi, linetype=:steppre, label="Viterbi decoded hidden state", size=(600,200))
